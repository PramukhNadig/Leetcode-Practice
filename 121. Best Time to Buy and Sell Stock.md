class Solution {
    public int maxProfit(int[] prices) {
        int minSoFar = 100000;
        int best = 0;
        
        for(int i = 0; i < prices.length; i++){
            
            if(prices[i] < minSoFar){
                minSoFar = prices[i];
            }
            if(prices[i] - minSoFar > best){
                best = prices[i] - minSoFar;
            }
        }
        
        return best;
    }
}
