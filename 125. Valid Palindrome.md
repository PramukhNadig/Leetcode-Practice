class Solution {
    public boolean isPalindrome(String s) {
        s = removeNonAlphaCharacters(s);
        
        int left = 0;
        int right = s.length() - 1;
        
        while(left <= right){
            if(s.charAt(left) != s.charAt(right)){
                return false;
            }
            
            left++;
            right--;
        }
        
        return true;
    }
    
    
    
    public String removeNonAlphaCharacters(String s){
        StringBuilder sb = new StringBuilder();
        s = s.toLowerCase();
        for(int i = 0; i < s.length(); i++){
            char c = s.charAt(i);
            if((c >= '0' && c <= '9') || (c >= 'a' && c <= 'z')){
                sb.append(c);
            }
        }
        return sb.toString();
    }
}
