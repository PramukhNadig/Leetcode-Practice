class MinStack {

    
    Stack<Integer> stack;
    PriorityQueue<Integer> heap;
    
    public MinStack() {
        this.stack = new Stack<Integer>();
        this.heap = new PriorityQueue<Integer>();
    }
    
    public void push(int val) {
        stack.push(val);
        heap.add(val);
    }
    
    public void pop() {        
        heap.remove(stack.pop());
    }
    
    public int top() {
        return stack.peek();
    }
    
    public int getMin() {
        return heap.peek();
    }
}
