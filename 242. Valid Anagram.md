class Solution {
    public boolean isAnagram(String s, String t) {
        int[] c = new int[256];
        
        for(int i = 0; i < s.length(); i++){
            c[s.charAt(i)]++;
        }
        
        
        for(int i = 0; i < t.length(); i++){
            c[t.charAt(i)]--;
        }
        
        
        for(int i = 0; i < 256; i++){
            if(c[i] != 0){
                return false;
            }
        }
        return true;
    }
}
